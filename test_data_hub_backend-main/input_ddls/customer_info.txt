CREATE TABLE customer_info (
    rim_no VARCHAR(24) PRIMARY KEY,
    rim_type VARCHAR(4),
    class_code VARCHAR(8),
    birth_date DATE,
    birth_city VARCHAR(32),
    country_code VARCHAR(8),
    gender CHAR(1),
    language VARCHAR(8),
    nationality VARCHAR(32),
    occupation VARCHAR(32),
    age INT,
    age_bracket VARCHAR(16),
    marital_status VARCHAR(16),
    resident_city VARCHAR(32),
    address VARCHAR(128),
    created_dt DATE,
    closed_dt DATE,
    status VARCHAR(12),
    employer_id VARCHAR(24),
    employer_name VARCHAR(64),
    employer_category VARCHAR(32),
    clean_employer_name VARCHAR(64),
    class_code_duplicate VARCHAR(8), -- redundant, can omit in production
    class_code_desc VARCHAR(32),
    business_unit VARCHAR(16),
    customer_segment VARCHAR(24),
    id_proof_type VARCHAR(16),
    id_proof_number VARCHAR(32),
    id_expiry_date DATE,
    kyc_verification_date DATE,
    kyc_status VARCHAR(16),
    is_kyc_verified BOOLEAN,
    mobile_app_flag CHAR(1),
    income_annual DECIMAL(20,2)
);
