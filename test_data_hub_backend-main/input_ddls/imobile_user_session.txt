CREATE TABLE imobile_user_session (
    session_id VARCHAR(32) PRIMARY KEY,
    customer_id VARCHAR(24),
    login_timestamp TIMESTAMP,
    logout_timestamp TIMESTAMP,
    login_channel VARCHAR(16),
    auth_method VARCHAR(16),
    login_status VARCHAR(16),
    failure_reason VARCHAR(64),
    device_id VARCHAR(32),
    device_type VARCHAR(16),
    os_version VARCHAR(24),
    app_version VARCHAR(16),
    location_lat_long VARCHAR(64),
    network_type VARCHAR(16),
    ip_address VARCHAR(45),
    session_duration_sec INT,
    is_first_time_login CHAR(1),
    biometric_enabled CHAR(1),
    push_notification_enabled CHAR(1),
    created_dt DATE,
    last_updated_dt TIMESTAMP,
    password_hash VARCHAR(128),
    mpin_hash VARCHAR(128)
);
